95void rtimvStats::mtxUL_calcStats()
96{
97
98 if(!m_regionChanged)
99 {
100 return;
101 }
102
103 if(m_imv == nullptr)
104 {
105 m_regionChanged = 0;
106 return; //no data.
107 }
108
109 if(m_calMutex == nullptr)
110 {
111 m_regionChanged = 0;
112 return; //no data.
113 }
114
115 //Get the cal data mutex
116 std::shared_lock<std::shared_mutex> lockCal(*m_calMutex);
117
118 //mutex lock here (trylock, exit and wait if can't get it)
119 std::unique_lock<std::mutex> lockData(m_dataMutex, std::try_to_lock);
120 if(!lockData.owns_lock()) return;
121
122 //After locking, verify that this is still the same size.
123 if(m_imv->nx() != m_nx || m_imv->ny() != m_ny)
124 {
125 return;
126 }
127
128 //copy data so we can be safe from changes to image memory
129 for(size_t i=m_x0; i<m_x1; ++i)
130 {
131 for(size_t j=m_y0; j<m_y1; ++j)
132 {
133 m_imdata(i-m_x0, j-m_y0) = m_imv->calPixel(i,j);
134 }
135 }
136
137 lockCal.unlock();
138
139 float tmp_min = m_imdata(0,0);
140 float tmp_max = tmp_min;
141 float tmp_mean = 0;
142
143 m_medWork.resize(m_imdata.cols()*m_imdata.rows()); //Should be a no-op most of the time
144 size_t nn = 0;
145 for(int c = 0; c < m_imdata.cols(); ++c)
146 {
147 for(int r = 0; r < m_imdata.rows(); ++r)
148 {
149 float imval = m_imdata(r,c);
150
151 tmp_mean += imval;
152 if(imval < tmp_min) tmp_min = imval;
153 if(imval > tmp_max) tmp_max = imval;
154
155 m_medWork[nn] = imval;
156 ++nn;
157 }
158 }
159
160 m_dataMin = tmp_min;
161 m_dataMax = tmp_max;
162 m_dataMean = tmp_mean / (m_imdata.rows()*m_imdata.cols());
163 m_dataMedian = mx::math::vectorMedianInPlace(m_medWork);
164
165 m_statsChanged = 1;
166 m_regionChanged = 0;
167
168}
