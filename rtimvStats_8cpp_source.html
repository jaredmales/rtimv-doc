112  //mutex lock here (trylock, exit and wait if can't get it)

115 
116  //copy data so we can be safe from changes to image memory
117  for(size_t i=m_x0; i<m_x1; ++i)
118  {
119  for(size_t j=m_y0; j<m_y1; ++j)
120  {
121  m_imdata(i-m_x0, j-m_y0) = m_imv->calPixel(i,j); //_pixel(m_imv, idx);
122  }
123  }
124 
125  lockCal.unlock();
126 
127  float tmp_min = m_imdata(0,0);
128  float tmp_max = tmp_min;
129  float tmp_mean = 0;
130 
131  for(int c = 0; c < m_imdata.cols(); ++c)
132  {
133  for(int r = 0; r < m_imdata.rows(); ++r)
134  {
135  float imval = m_imdata(r,c);
136
