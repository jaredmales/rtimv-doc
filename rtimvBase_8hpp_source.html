571template<typename lockT>
572void rtimvBase::mtxL_load_colorbar( int cb,
573 bool update,
574 const lockT & lock
575 )
576{
577 assert(lock.owns_lock());
578
579 if(!m_qim)
580 {
581 return;
582 }
583
584 current_colorbar = cb;
585 switch (cb)
586 {
587 case colorbarJet:
588 m_minColor = 0;
589 m_maxColor = load_colorbar_jet(m_qim);
590 m_maskColor = m_maxColor + 1;
591 m_satColor = m_maxColor + 2;
592 m_nanColor = m_maskColor;
593 warning_color = QColor("white");
594 break;
595 case colorbarHot:
596 m_minColor = 0;
597 m_maxColor = load_colorbar_hot(m_qim);
598 m_maskColor = m_maxColor + 1;
599 m_satColor = m_maxColor + 2;
600 m_nanColor = m_maskColor;
601 warning_color = QColor("cyan");
602 break;
603 case colorbarBone:
604 m_minColor = 0;
605 m_maxColor = load_colorbar_bone(m_qim);
606 m_maskColor = m_maxColor + 1;
607 m_satColor = m_maxColor + 2;
608 m_nanColor = m_maskColor;
609 warning_color = QColor("red");
610 break;
611 case colorbarRed:
612 m_minColor = 0;
613 m_maxColor = 253;
614 m_maskColor = m_maxColor + 1;
615 m_satColor = m_maxColor + 2;
616 m_nanColor = m_maskColor;
617 for(int i = m_minColor; i <= m_maxColor; i++)
618 {
619 int c = (((float)i) / 253. * 255.) + 0.5;
620 m_qim->setColor(i, qRgb(c, 0, 0));
621 }
622 m_qim->setColor(254, qRgb(0, 0, 0));
623 m_qim->setColor(255, qRgb(0, 255, 0));
624 warning_color = QColor("red");
625 break;
626 case colorbarGreen:
627 m_minColor = 0;
628 m_maxColor = 253;
629 m_maskColor = m_maxColor + 1;
630 m_satColor = m_maxColor + 2;
631 m_nanColor = m_maskColor;
632 for(int i = m_minColor; i <= m_maxColor; i++)
633 {
634 int c = (((float)i) / 253. * 255.) + 0.5;
635 m_qim->setColor(i, qRgb(0, c, 0));
636 }
637 m_qim->setColor(254, qRgb(0, 0, 0));
638 m_qim->setColor(255, qRgb(255, 0, 0));
639 warning_color = QColor("red");
640 break;
641 case colorbarBlue:
642 m_minColor = 0;
643 m_maxColor = 253;
644 m_maskColor = m_maxColor + 1;
645 m_satColor = m_maxColor + 2;
646 m_nanColor = m_maskColor;
647 for(int i = m_minColor; i <= m_maxColor; i++)
648 {
649 int c = (((float)i) / 253. * 255.) + 0.5;
650 m_qim->setColor(i, qRgb(0, 0, c));
651 }
652 m_qim->setColor(254, qRgb(0, 0, 0));
653 m_qim->setColor(255, qRgb(255, 0, 0));
654 warning_color = QColor("red");
655 break;
656 default:
657 m_minColor = 0;
658 m_maxColor = 253;
659 m_maskColor = m_maxColor + 1;
660 m_satColor = m_maxColor + 2;
661 m_nanColor = m_maskColor;
662 for(int i = m_minColor; i <= m_maxColor; i++)
663 {
664 int c = (((float)i) / 253. * 255.) + 0.5;
665 m_qim->setColor(i, qRgb(c, c, c));
666 }
667 m_qim->setColor(254, qRgb(0, 0, 0));
668 m_qim->setColor(255, qRgb(255, 0, 0));
669
670 warning_color = QColor("red");
671 break;
672 }
673
674 m_lightness.resize(256);
675
676 for(int n = 0; n < 256; ++n)
677 {
678 m_lightness[n] = QColor(m_qim->color(n)).lightness();
679 }
680
681 if(update)
682 {
683 mtxL_recolor(lock);
684 }
685}
