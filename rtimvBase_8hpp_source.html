669template <typename lockT>
670void rtimvBase::mtxL_load_colorbar( int cb, bool update, const lockT &lock )
671{
672 assert( lock.owns_lock() );
673
674 if( !m_qim )
675 {
676 return;
677 }
678
679 current_colorbar = cb;
680 switch( cb )
681 {
682 case colorbarJet:
683 m_minColor = 0;
684 m_maxColor = load_colorbar_jet( m_qim );
685 m_maskColor = m_maxColor + 1;
686 m_satColor = m_maxColor + 2;
687 m_nanColor = m_maskColor;
688 warning_color = QColor( "white" );
689 break;
690 case colorbarHot:
691 m_minColor = 0;
692 m_maxColor = load_colorbar_hot( m_qim );
693 m_maskColor = m_maxColor + 1;
694 m_satColor = m_maxColor + 2;
695 m_nanColor = m_maskColor;
696 warning_color = QColor( "cyan" );
697 break;
698 case colorbarBone:
699 m_minColor = 0;
700 m_maxColor = load_colorbar_bone( m_qim );
701 m_maskColor = m_maxColor + 1;
702 m_satColor = m_maxColor + 2;
703 m_nanColor = m_maskColor;
704 warning_color = QColor( "red" );
705 break;
706 case colorbarRed:
707 m_minColor = 0;
708 m_maxColor = 253;
709 m_maskColor = m_maxColor + 1;
710 m_satColor = m_maxColor + 2;
711 m_nanColor = m_maskColor;
712 for( int i = m_minColor; i <= m_maxColor; i++ )
713 {
714 int c = ( ( (float)i ) / 253. * 255. ) + 0.5;
715 m_qim->setColor( i, qRgb( c, 0, 0 ) );
716 }
717 m_qim->setColor( 254, qRgb( 0, 0, 0 ) );
718 m_qim->setColor( 255, qRgb( 0, 255, 0 ) );
719 warning_color = QColor( "red" );
720 break;
721 case colorbarGreen:
722 m_minColor = 0;
723 m_maxColor = 253;
724 m_maskColor = m_maxColor + 1;
725 m_satColor = m_maxColor + 2;
726 m_nanColor = m_maskColor;
727 for( int i = m_minColor; i <= m_maxColor; i++ )
728 {
729 int c = ( ( (float)i ) / 253. * 255. ) + 0.5;
730 m_qim->setColor( i, qRgb( 0, c, 0 ) );
731 }
732 m_qim->setColor( 254, qRgb( 0, 0, 0 ) );
733 m_qim->setColor( 255, qRgb( 255, 0, 0 ) );
734 warning_color = QColor( "red" );
735 break;
736 case colorbarBlue:
737 m_minColor = 0;
738 m_maxColor = 253;
739 m_maskColor = m_maxColor + 1;
740 m_satColor = m_maxColor + 2;
741 m_nanColor = m_maskColor;
742 for( int i = m_minColor; i <= m_maxColor; i++ )
743 {
744 int c = ( ( (float)i ) / 253. * 255. ) + 0.5;
745 m_qim->setColor( i, qRgb( 0, 0, c ) );
746 }
747 m_qim->setColor( 254, qRgb( 0, 0, 0 ) );
748 m_qim->setColor( 255, qRgb( 255, 0, 0 ) );
749 warning_color = QColor( "red" );
750 break;
751 default:
752 m_minColor = 0;
753 m_maxColor = 253;
754 m_maskColor = m_maxColor + 1;
755 m_satColor = m_maxColor + 2;
756 m_nanColor = m_maskColor;
757 for( int i = m_minColor; i <= m_maxColor; i++ )
758 {
759 int c = ( ( (float)i ) / 253. * 255. ) + 0.5;
760 m_qim->setColor( i, qRgb( c, c, c ) );
761 }
762 m_qim->setColor( 254, qRgb( 0, 0, 0 ) );
763 m_qim->setColor( 255, qRgb( 255, 0, 0 ) );
764
765 warning_color = QColor( "red" );
766 break;
767 }
768
769 m_lightness.resize( 256 );
770
771 for( int n = 0; n < 256; ++n )
772 {
773 m_lightness[n] = QColor( m_qim->color( n ) ).lightness();
774 }
775
776 if( update )
777 {
778 mtxL_recolor( lock );
779 }
780}
