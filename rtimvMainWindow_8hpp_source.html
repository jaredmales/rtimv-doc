119 /** \name North Arrow
120 * The north arrow is toggled on/off with the `n` key. The angle of the north arrow returned by northAngle() is determined by
121 * the dictionary key "rtimv.north.angle", which can be set from a dictionary plugin or using the slot
122 * northAngleRaw(float). This is multiplied by m_northAngleScale and then m_northAngleOffset is added to the result to produce
123 * a counter clockwise rotation angle in degrees.
124 *
125 * The north arrow can be disabled in configuration. If "north.enabled=false" is set, the north arrow will not be displayed.
126 *
127 * Relevant configuration key=value pairs are

129 * - `north.offset=float`: Offset in degrees c.c.w. to apply to the north angle. Default is 0.

695protected:
696 /// The power for averaging pixel luminance.
697 /** The average is formed as pow( sum(pow(pixel-luminance, m_lumPwr), 1/m_lumPwr). This
698 * being a large number causes a few very bright pixels to bias the average.
699 */
700 double m_lumPwr{12};
701
702 /// The luminance threshold below which no background transparency is set.
703 /** Above this, but below m_lumMax, linear interpolation is used to set the opacity.
704 *
705 * Range: 0-255;
706 */
707 double m_lumThresh{100};
708
709 /// The maximum luminance, at which point the opacity is set to m_opacityMax.
710 /** Below this, but above m_lumThresh, linear interpolation is used to set the opacity.
711 *
712 * Range: 0-255;
713 */
714 double m_lumMax{175};
715
716 /// The maximum opacity which will be set.
717 /** This is the opacity set when at or above m_lumMax.
718 * Between m_lumThresh and m_lumMax, the opacity is interpolated between 0 and this value.
