91bool StretchBox::onHoverComputeSizing( QGraphicsSceneHoverEvent *e )
92{
93 double rx, ry;
94
95 rx = e->lastPos().x() - rect().x();
96 ry = e->lastPos().y() - rect().y();
97
98 if( rx > m_edgeTol && rx < rect().width() - m_edgeTol && ry > m_edgeTol && ry < rect().height() - m_edgeTol )
99 {
100 return false;
101 }
102
103 // Upper Left Corner
104 if( rx > -m_edgeTol && rx < m_edgeTol && ry > -m_edgeTol && ry < m_edgeTol )
105 {
106 m_sizing = szTopl;
107 }
108 else if( rx > -m_edgeTol && rx < m_edgeTol && ry > rect().height() - m_edgeTol && ry < rect().height() + m_edgeTol )
109 {

111 }
112 else if( rx > rect().width() - m_edgeTol && rx < rect().width() + m_edgeTol && ry > -m_edgeTol && ry < m_edgeTol )

179void StretchBox::sizingCalcNewPos( QGraphicsSceneMouseEvent *e )
180{
181 double newx, newy, neww, newh, dw, dh;
182
183 dw = ( e->scenePos().x() - m_mv_x0 );
184 dh = ( e->scenePos().y() - m_mv_y0 );
185
186 if( m_sizing == szLeft )
187 {
188 newx = m_ul_x + dw;
189 if( m_maintainCenter )
190 {
191 dw *= 2;
192 }
193
194 newy = m_ul_y;
195 neww = m_width - dw;
196 newh = m_height;
197 }
198 else if( m_sizing == szTopl )
199 {
200 newx = m_ul_x + dw;
201 newy = m_ul_y + dh;
202
203 if( m_maintainCenter )
204 {
205 dw *= 2;
206 dh *= 2;
207 }
208
209 neww = m_width - dw;
210 newh = m_height - dh;
211 }
212 else if( m_sizing == szTop )
213 {
214 newx = m_ul_x;
215 newy = m_ul_y + dh;
216 if( m_maintainCenter )
217 {
218 dh *= 2;
219 }
220
221 neww = m_width;
222 newh = m_height - dh;
223 }
224 else if( m_sizing == szTopr )
225 {
226 newx = m_ul_x;
227 newy = m_ul_y + dh;
228 if( m_maintainCenter )
229 {
230 newx -= dw;
231
232 dw *= 2;
233 dh *= 2;
234 }
235
236 neww = m_width + dw;
237 newh = m_height - dh;
238 }
239 else if( m_sizing == szRight )
240 {
241 newx = m_ul_x;
242 newy = m_ul_y;
243

246 newx -= dw;
