705 {

707  {

719

722  {

724  setImsize(m_images[0]->nx(), m_images[0]->ny());

733 
734  // Get the pixel calculating function

742  }

745  {
746  m_amChangingimdata = false;
747  return;
748  }

750  for(uint64_t n=0; n < m_nx*m_ny; ++n)
