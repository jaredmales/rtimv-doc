83bool StretchLine::onHoverComputeSizing(QGraphicsSceneHoverEvent * e)
84{
85 float x = e->pos().x() - line().x1();
86 float y = e->pos().y() - line().y1();
87
88 float m, drad;
89 if(line().x2()-line().x1() == 0)
90 {
91 m = 1e50;
92 drad = x;
93 }
94 else
95 {
96 m = (line().y2()-line().y1())/(line().x2()-line().x1());
97 drad = fabs(-m*x + y)/sqrt(m*m+1);
98 }
99
100 if(drad > m_edgeTol)
101 {
102 return false;
103 }
104
105
106 float r1 = sqrt(x*x + y*y);
107
108 float x2 = e->pos().x() - line().x2();
109 float y2 = e->pos().y() - line().y2();
110
111 float r2 = sqrt(x2*x2 + y2*y2);
112 float l = line().length();
113 if( r2 < m_edgeTol || r2 < 0.1*l) //First make sure we can always change length
114 {
115 m_sizing = szAll;
116 }
117 else if( r1 < m_edgeTol || r1 < 0.2*l) //Don't allow rotate if too close to the 0
118 {
119 m_sizing = szOff;
120 return true;
121 }
122 else
123 {
124 float ang = line().angle();
125 if(ang >= 347.4 || ang <= 22.5 || (ang >= 157.5 && ang <=202.5) ) m_sizing = szTop;
126 else if( (ang >= 22.5 && ang <= 67.5) || (ang >= 202.5 && ang <= 247.5) ) m_sizing = szBotr;
127 else if( (ang >= 67.5 && ang <= 112.5) || (ang >= 247.5 && ang <= 292.5) ) m_sizing = szLeft;
128 else m_sizing = szBotl;
129
130 }
131
132 return true;
133}

189void StretchLine::sizingCalcNewPos(QGraphicsSceneMouseEvent * e)
190{
191 if(m_sizing == szAll)
192 {
193 float dx = e->pos().x() - m_mv_x0;
194 float dy = e->pos().y() - m_mv_y0;
195
196 setLine(m_start_x, m_start_y, m_end_x+dx, m_end_y+dy);
197 }
198 else
199 {
200 float dang = -(QLineF(m_start_x, m_start_y, e->pos().x(), e->pos().y()).angle() - m_mv_ang0);
201
202 float c = cos(dang*3.14159/180.);
203 float s = sin(dang*3.14159/180.);
204
205 float dx = (m_end_x - m_start_x)*c - (m_end_y-m_start_y)*s;
206 float dy = (m_end_x - m_start_x)*s + (m_end_y-m_start_y)*c;
207
208 setLine(m_start_x, m_start_y, m_start_x+dx, m_start_y+dy);
209 }
