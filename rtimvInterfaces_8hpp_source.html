95 /// Set the blob
96 /** Allocates m_blob and copies \p sz bytes from \p blob.
97 * Allocation only occurs if more size is needed according to m_memSz.
98 * Sets m_sz to sz.
99 */
100 void setBlob( const void *blob, ///< [in] The new blob data
101 size_t sz ///< [in] The number of bytes in the new blob data
102 )
103 {
104 std::unique_lock<std::mutex> lock( m_mutex );
105 if( sz > m_memSz || m_owner == false )
106 {
107 clearBlob();
108
109 if( sz > 0 )
110 {
111 m_blob = malloc( sz );
112 m_sz = sz;
113 m_memSz = sz;
114 m_owner = true;
115 }
116
117 if( blob != nullptr && sz > 0 )
118 {
119 memcpy( m_blob, blob, sz );
120 }
121 }
122 else
123 {
124 // We're the owner and not reallocating
125 m_sz = sz;
126 if( blob != nullptr && sz > 0 )
127 {
128 memcpy( m_blob, blob, sz );
129 }
130 }
131
132 clock_gettime( CLOCK_REALTIME, &m_lastMod );
133 }
134
135 /// Get the data stored in the blob
136 /** Copies the smaller of \p mxsz and m_sz bytes to \p blob
137 * \returns the number of bytes copied
138 */
139 size_t
140 getBlob( char *blob, ///< [out] The pointer to the array to copy the data to
141 size_t mxsz, ///< [in] The max number of bytes to copy to blob
142 timespec *ts =
143 nullptr /// [out] [optional] if not null this will be filled in with the time of last modification
144 )
145 {
146 std::unique_lock<std::mutex> lock( m_mutex );
147
148 size_t cpsz = m_sz;
149 if( cpsz > mxsz )
150 cpsz = mxsz;
151
152 if( m_blob )
153 {
154 memcpy( blob, m_blob, cpsz );
155 }
156 else
157 cpsz = 0;
158
159 if( ts != nullptr )
160 *ts = m_lastMod;
161
162 return cpsz;
163 }
164
165 /// Get the data stored in the blob as a C string
166 /** Copies the smaller of \p mxsz and m_sz bytes to \p blob.
167 *
168 * This guarantees that blob[sz-1] = 0
169 * where sz is the number of bytes copied. If the returned size == mxsz then you should not trust the
170 * contents unless you know that the correct size was copied or don't care if the string is truncated.
171 *
172 * If m_blob has 0 size or is nullptr, then this sets blob[0] = 0 and returns 1.
173 *
174 * \returns the number of bytes copied
175 */
176 size_t
177 getBlobStr( char *blob, ///< [in/out] The pointer to the array to copy the data to
178 size_t mxsz, ///< [in] The max number of bytes to copy to blob
179 timespec *ts =
180 nullptr /// [out] [optional] if not null this will be filled in with the time of last modification
181 )
182 {
183 size_t cpsz = getBlob( blob, mxsz, ts );
184
185 if( cpsz == 0 )
186 {
187 if( mxsz > 0 )
188 {
189 blob[0] = '\0';
190 cpsz = 1;
191 }
192 }
193 else
194 blob[cpsz - 1] = '\0';
195
196 return cpsz;
197 }

199 size_t getBlobSize()

204 return 0;
205 return m_sz;
