619 void rtimvGraphicsView::showMouseToolTip( const std::string & valStr,
620  const std::string & posStr,
621  const QPoint & pt
622  )
623 {
624  std::string str;
625 
626  m_mouseCoords->setWordWrapMode(QTextOption::NoWrap);
627 
628  int x = pt.x();
629  int y = pt.y();
630 
631  //First construct the text in correct order for y position
632  if(y > this->height()*0.5)
633  {
634  std::string sp;
635  if(x > this->width()*0.5)
636  {
637  //Right justify the value string if needed
638  if(posStr.size() > valStr.size())
639  {
640  sp.append(posStr.size() - valStr.size(), ' ');
641  }
642  }
643  str = posStr + "\n" + sp + valStr;
644  }
645  else
646  {
647  str = valStr + "\n" + posStr;
648  }
649 
650  m_mouseCoords->setPlainText(str.c_str());
651 
652  QFontMetrics fm(m_mouseCoords->currentFont());
653  QSize textSize = fm. size(0, str.c_str());
654  m_mouseCoords->resize(textSize.width()+5, textSize.height());
655 
656  if(x > this->width()*0.5)
657  {
658  x -= m_mouseCoords->width();
659  if(x < 0) x = 0;
660  m_mouseCoords->setAlignment(Qt::AlignRight);
661  }
662  else
663  {
664  if(y < this->height()*0.5) x+=10; //have to move past the arrow
665  m_mouseCoords->setAlignment(Qt::AlignLeft);
666  }
667 
668  if(y > this->height()*0.5)
669  {
670  y-= m_mouseCoords->height();
671  }
672 
673  m_mouseCoords->move(QPoint(x,y));
674  m_mouseCoords->show();
675 
676 }
