635  const std::string & posStr,

637  )

642

644  int y = pt.y();
645 
646  //First construct the text in correct order for y position
647  if(y > this->height()*0.5)
648  {
649  std::string sp;
650  if(x > this->width()*0.5)
651  {
652  //Right justify the value string if needed
653  if(posStr.size() > valStr.size())
654  {
655  sp.append(posStr.size() - valStr.size(), ' ');
656  }
657  }
658  str = posStr + "\n" + sp + valStr;
659  }
660  else
661  {
662  str = valStr + "\n" + posStr;
663  }
664

666 
667  QFontMetrics fm(m_mouseCoords->currentFont());
668  QSize textSize = fm. size(0, str.c_str());
669  m_mouseCoords->resize(textSize.width()+5, textSize.height());
670 
671  if(x > this->width()*0.5)
672  {
673  x -= m_mouseCoords->width();
674  if(x < 0) x = 0;
675  m_mouseCoords->setAlignment(Qt::AlignRight);
676  }
677  else
678  {
679  if(y < this->height()*0.5) x+=10; //have to move past the arrow
680  m_mouseCoords->setAlignment(Qt::AlignLeft);
681  }
682 
683  if(y > this->height()*0.5)
684  {
685  y-= m_mouseCoords->height();
686  }
687 
688  m_mouseCoords->move(QPoint(x,y));
