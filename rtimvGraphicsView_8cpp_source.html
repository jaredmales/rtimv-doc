616void rtimvGraphicsView::showMouseToolTip( const std::string &valStr, const std::string &posStr, const QPoint &pt )
617{
618 std::string str;
619
620 int x = pt.x();
621 int y = pt.y();
622
623 // First construct the text in correct order for y position
624 if( y > this->height() * 0.5 )
625 {
626 std::string sp;
627 if( x > this->width() * 0.5 )
628 {
629 // Right justify the value string if needed
630 if( posStr.size() > valStr.size() )
631 {
632 sp.append( posStr.size() - valStr.size(), ' ' );
633 }
634 }
635 str = posStr + "\n" + sp + valStr;
636 }
637 else
638 {
639 str = valStr + "\n" + posStr;
640 }
641
642 m_mouseCoords->setPlainText( str.c_str() );
643
644 QFontMetrics fm( m_mouseCoords->currentFont() );
645 QSize textSize = fm.size( 0, str.c_str() );
646 m_mouseCoords->resize( textSize.width() + 5, textSize.height() );
647
648 if( x > this->width() * 0.5 )
649 {
650 x -= m_mouseCoords->width();
651 if( x < 0 )
652 x = 0;
653 m_mouseCoords->setAlignment( Qt::AlignRight );
654 }
655 else
656 {
657 if( y < this->height() * 0.5 )
658 x += 10; // have to move past the arrow
659 m_mouseCoords->setAlignment( Qt::AlignLeft );
660 }
661
662 if( y > this->height() * 0.5 )
663 {
664 y -= m_mouseCoords->height();
665 }
666
667 m_mouseCoords->move( QPoint( x, y ) );
668 m_mouseCoords->show();
669}

676void rtimvGraphicsView::textPixelVal( float nv )
677{
678 // 1.00e+5
679 // 10000.0
680 // 1000.0
681 // 100.0
682 // 10.0
683 // 1.0
684 // 0.10
685 // 0.010
686 // 0.0010
687 // 0.00010
688 // 1.0e-3
689 char strtmp[24];
690
691 float anv = fabs( nv );
692 if( anv == 0 )
693 {
694 snprintf( strtmp, sizeof( strtmp ), "Z: 0" );
695 }
696 else if( anv >= 100000 )
697 {
698 snprintf( strtmp, sizeof( strtmp ), "Z: %0.2e", nv );
699 }
700 else if( anv >= 1.0 )
701 {
702 snprintf( strtmp, sizeof( strtmp ), "Z: %0.1f", nv );
703 }
704 else if( anv >= 0.1 )
705 {
706 snprintf( strtmp, sizeof( strtmp ), "Z: %0.2f", nv );
707 }
708 else if( anv >= 0.01 )
709 {
710 snprintf( strtmp, sizeof( strtmp ), "Z: %0.3f", nv );
711 }
712 else if( anv >= 0.001 )
713 {
714 snprintf( strtmp, sizeof( strtmp ), "Z: %0.4f", nv );
715 }
716 else if( anv >= 0.0001 )
717 {
718 snprintf( strtmp, sizeof( strtmp ), "Z: %0.5f", nv );
719 }
720 else
721 {
722 snprintf( strtmp, sizeof( strtmp ), "Z: %0.1e", nv );
723 }
724
725 textPixelVal( strtmp );
726}
